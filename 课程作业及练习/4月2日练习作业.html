变量声明

JavaScript中有三种声明变量的方式：

  • var - 函数作用域
  • let - 块级作用域
  • const - 块级作用域常量
var name = "张三"; // 函数作用域
let age = 25; // 块级作用域
const PI = 3.14; // 常量，不可重新赋值

// 变量命名规则
let firstName = "李"; // 驼峰命名法
let $price = 99.99; // 允许$开头
let _internal = "secret"; // 允许_开头

数据类型

JavaScript有7种原始数据类型和对象类型：

  • String - 字符串
  • Number - 数字
  • Boolean - 布尔值
  • Null - 空值
  • Undefined - 未定义
  • Symbol - 符号(ES6)
  • BigInt - 大整数(ES2020)
  • Object - 对象
typeof "Hello"; // "string"
typeof 42; // "number"
typeof true; // "boolean"
typeof undefined; // "undefined"
typeof null; // "object" (历史遗留问题)
typeof {name: "John"}; // "object"
typeof [1, 2, 3]; // "object"
typeof function() {}; // "function"

函数

JavaScript中定义函数的几种方式：

// 函数声明
function add(a, b) {
  return a + b;
}

// 函数表达式
const subtract = function(a, b) {
  return a - b;
};

// 箭头函数 (ES6)
const multiply = (a, b) => a * b;

// 立即执行函数表达式 (IIFE)
(function() {
  console.log('立即执行!');
})();

// 函数参数默认值
function greet(name = '访客') {
  return `你好, ${name}!`;
}

对象

对象是键值对的集合，是JavaScript的核心概念。

// 对象字面量
const person = {
  firstName: '张',
  lastName: '三',
  age: 30,
  isStudent: false,
  hobbies: ['阅读', '旅行', '编程'],
  address: {
    city: '北京',
    country: '中国'
  },
  fullName: function() {
    return this.firstName + ' ' + this.lastName;
  }
};

// 访问属性
console.log(person.firstName); // "张"
console.log(person['lastName']); // "三"
console.log(person.fullName()); // "张三"

// ES6 简写语法
const name = '李四';
const age = 28;
const user = { name, age }; // {name: "李四", age: 28}

数组

数组是用于存储多个值的有序集合。

// 创建数组
let fruits = ['苹果', '香蕉', '橙子'];
let numbers = new Array(1, 2, 3);

// 常用数组方法
fruits.push('芒果'); // 末尾添加
fruits.pop(); // 移除最后一个
fruits.shift(); // 移除第一个
fruits.unshift('葡萄'); // 开头添加

// ES6+ 数组方法
const doubled = numbers.map(num => num * 2); // [2, 4, 6]
const even = numbers.filter(num => num % 2 === 0); // [2]
const sum = numbers.reduce((total, num) => total + num, 0); // 6

// 解构赋值
const [first, second] = fruits; // first='葡萄', second='香蕉'
const [a, b, ...rest] = [1, 2, 3, 4, 5]; // a=1, b=2, rest=[3,4,5]

异步编程

JavaScript处理异步操作的几种方式：

// 回调函数
function fetchData(callback) {
  setTimeout(() => {
    callback('数据已获取');
  }, 1000);
}

// Promise
const promise = new Promise((resolve, reject) => {
  setTimeout(() => {
    const success = true;
    success ? resolve('成功!') : reject('失败!');
  }, 1500);
});

promise
  .then(result => console.log(result))
  .catch(error => console.log(error));

// Async/Await
async function getData() {
  try {
    const response = await fetch('https://api.example.com/data');
    const data = await response.json();
    console.log(data);
  } catch (error) {
    console.log('错误:', error);
  }
}
getData();

DOM操作

使用JavaScript操作文档对象模型(DOM)：

// 选择元素
const element = document.getElementById('myId');
const elements = document.querySelectorAll('.myClass');

// 修改内容
element.textContent = '新文本内容';
element.innerHTML = '<strong>加粗文本</strong>';

// 修改样式
element.style.color = 'blue';
element.style.backgroundColor = '#f0f0f0';

// 添加/移除类
element.classList.add('new-class');
element.classList.remove('old-class');
element.classList.toggle('active');

// 创建新元素
const newElement = document.createElement('div');
newElement.textContent = '新创建的元素';
document.body.appendChild(newElement);
