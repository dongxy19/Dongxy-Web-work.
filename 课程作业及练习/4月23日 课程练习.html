变量与数据类型

JavaScript中的变量声明和基本数据类型

变量声明

JavaScript中有三种声明变量的方式：var, let, const

// 使用var声明变量 (函数作用域)
var name = "张三";

// 使用let声明变量 (块级作用域)
let age = 25;

// 使用const声明常量 (不可重新赋值)
const PI = 3.14159;

数据类型

JavaScript有7种基本数据类型和1种复杂数据类型

// 基本数据类型
let str = "字符串"; // String
let num = 42; // Number
let bool = true; // Boolean
let undef; // Undefined
let empty = null; // Null
let sym = Symbol('unique'); // Symbol
let big = 12345678901234567890n; // BigInt

// 复杂数据类型
let obj = { name: "李四", age: 30 }; // Object

函数

JavaScript中函数的定义和使用

函数声明

JavaScript中有多种定义函数的方式

// 函数声明
function add(a, b) {
  return a + b;
}

// 函数表达式
const multiply = function(a, b) {
  return a * b;
};

// 箭头函数 (ES6)
const divide = (a, b) => a / b;

// 立即执行函数表达式 (IIFE)
(function() {
  console.log("立即执行");
})();

对象

JavaScript对象的创建和操作

对象创建

创建对象的多种方式

// 对象字面量
const person = {
  name: "王五",
  age: 28,
  greet: function() {
    console.log(`你好，我是${this.name}`);
  }
};

// 使用构造函数
function Car(brand, year) {
  this.brand = brand;
  this.year = year;
}
const myCar = new Car("Toyota", 2020);

// 使用Object.create()
const student = Object.create(null);
student.name = "赵六";
student.grade = "A";

数组

JavaScript数组的创建和常用方法

数组操作

数组的创建和常用操作方法

// 创建数组
const fruits = ['苹果', '香蕉', '橙子'];

// 访问元素
const first = fruits[0]; // '苹果'

// 添加/删除元素
fruits.push('芒果'); // 添加到末尾
fruits.pop(); // 删除最后一个元素
fruits.unshift('草莓'); // 添加到开头
fruits.shift(); // 删除第一个元素

// 数组遍历方法
fruits.forEach(function(fruit) {
  console.log(fruit);
});

// map方法 (返回新数组)
const upperFruits = fruits.map(fruit => fruit.toUpperCase());

流程控制

条件语句和循环结构

条件语句

根据条件执行不同的代码块

// if...else语句
let score = 85;
if (score >= 90) {
  console.log("优秀");
} else if (score >= 80) {
  console.log("良好");
} else {
  console.log("继续努力");
}

// switch语句
let day = "周一";
switch(day) {
  case "周一":
    console.log("工作日开始");
    break;
  case "周五":
    console.log("周末快到了");
    break;
  default:
    console.log("普通工作日");
}

循环结构

重复执行代码块

// for循环
for (let i = 0; i < 5; i++) {
  console.log(i); // 输出0,1,2,3,4
}

// while循环
let j = 0;
while (j < 5) {
  console.log(j);
  j++;
}

// for...of循环 (遍历数组)
const colors = ['红', '绿', '蓝'];
for (const color of colors) {
  console.log(color);
}

// for...in循环 (遍历对象属性)
const user = {name: "张三", age: 30};
for (const key in user) {
  console.log(`${key}: ${user[key]}`);
}
